import tkinter as tk
from tkinter import *
from functools import partial
import mysql.connector

login = Tk()	# okno do zalogowania
login.geometry('800x500') 
login.title('DebtyVibe')
window_width = 800
window_height = 500
screen_width = login.winfo_screenwidth() 	# ustawianie okna logowania
screen_height = login.winfo_screenheight()
x = (screen_width - window_width) // 2
y = (screen_height - window_height) // 6
login.geometry('{}x{}+{}+{}'.format(window_width, window_height, x, y))


# Zaloguj się

signin_frame = LabelFrame(login, text="ZALOGUJ SIĘ",font=("times new roman",15,"bold"),bg="grey",fg="white",bd=5,relief=GROOVE)
signin_frame.place(x=0,y=0,width=400,height=500)

def signin(username, password):	# funkcja do zalogowania sie
	print("username entered :", username.get())
	print("password entered :", password.get())
	return

usernameLabel = Label(login, text="User Name").place(x=150, y=150)
username = StringVar()
usernameEntry = Entry(login, textvariable=username).place(x=150, y=200)

passwordLabel = Label(login,text="Password").place(x=150, y=250)
password = StringVar()
passwordEntry = Entry(login, textvariable=password, show='*').place(x=150, y=300)

validateLogin = partial(signin, username, password)

loginButton = Button(login, text="Login", command=validateLogin).pack()

def sign_in():
	login.destroy()
	from MusicPlayer import main

btn = Button(text="idk", command=sign_in).pack()

# Utwórz konto

signup_frame = LabelFrame(login, text="UTWÓRZ KONTO",font=("times new roman",15,"bold"),bg="grey",fg="white",bd=5,relief=GROOVE)
signup_frame.place(x=400,y=0,width=400,height=500)

def signout(username, password):	# funkcja do stworzenia konta
	print("username entered :", username.get())
	print("password entered :", password.get())
	return

usernameLabel = Label(login, text="User Name").place(x=150, y=150)
username = StringVar()
usernameEntry = Entry(login, textvariable=username).place(x=150, y=150)

passwordLabel = Label(login,text="Password").place(x=150, y=150)
password = StringVar()
passwordEntry = Entry(login, textvariable=password, show='*').place(x=150, y=150)

validateLogin = partial(validateLogin, username, password)

loginButton = Button(login, text="Login", command=validateLogin).place(x=150, y=150)

def sign_out():
	login.destroy()
	from MusicPlayer import main

btn = Button(text="idk", command=sign_out).pack()

login.mainloop()
